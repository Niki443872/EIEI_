rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // Simple auth check
    function isSignedIn() {
      return request.auth != null;
    }

    // Check if user is admin
    function isAdmin() {
      let userEmail = request.auth.token.email;
      let userRole = get(/databases/$(database)/documents/roles/$(userEmail)).data;
      return userRole != null && 
             (userRole.role == 'admin' || userRole.role == 'superadmin');
    }

    // Check if user is superadmin
    function isSuperAdmin() {
      let userEmail = request.auth.token.email;
      let userRole = get(/databases/$(database)/documents/roles/$(userEmail)).data;
      return userRole != null && userRole.role == 'superadmin';
    }

    // --- CHAT SYSTEM (NEW) ---
    match /chats/{chatId} {
      // Admins (authenticated) can read and write all conversations
      allow read, write: if isSignedIn() && isAdmin();
      
      // Anyone can create a new chat
      allow create: if true;
      
      // Allow visitors to read and update their own chats (unauthenticated)
      allow read, update, delete: if request.auth == null;
      
      match /messages/{messageId} {
        // Admins can read and write all messages
        allow read, write: if isSignedIn() && isAdmin();
        
        // Unauthenticated visitors can read all messages
        allow read: if request.auth == null;
        
        // Unauthenticated visitors can create messages with sender='user'
        allow create: if request.auth == null && request.resource.data.sender == 'user';
        
        // Unauthenticated visitors can update messages (for marking as read)
        allow update: if request.auth == null;
      }
    }

    // Helper to get parent document
    function getParent(level) {
      return get(/databases/$(database)/documents/$(path(level)));
    }

    // --- Public site content ---
    match /posts/{id} {
      allow read: if true;
      allow create, update, delete: if isSignedIn() && isAdmin();
    }

    match /programs/{id} {
      allow read: if true;
      allow create, update, delete: if isSignedIn() && isAdmin();
    }

    match /team/{id} {
      allow read: if true;
      allow create, update, delete: if isSignedIn() && isAdmin();
    }

    match /testimonials/{id} {
      allow read: if true;
      allow create, update, delete: if isSignedIn() && isAdmin();
    }

    // --- Newsletter subscriptions ---
    match /newsletter/{email} {
      allow read: if true;
      allow create: if request.resource.data.email is string && request.resource.id == request.resource.data.email && !exists(/databases/$(database)/documents/newsletter/$(request.resource.id));
      allow update, delete: if isSignedIn() && isAdmin();
    }

    // --- Enquiries (public form) ---
    match /enquiries/{id} {
      allow create: if true;
      allow read, update, delete: if isSignedIn() && isAdmin();
    }

    // --- Contacts (Contact Us form) ---
    match /contacts/{id} {
      allow create: if true;
      allow read, update, delete: if isSignedIn() && isAdmin();
    }

    // --- Marketplace items ---
    match /marketplace/{id} {
      allow read: if true;
      allow create, update, delete: if isSignedIn() && isAdmin();
    }

    // --- Roles & Permissions ---
    match /roles/{userEmail} {
      // Users can read their own role
      allow read: if isSignedIn() && request.auth.token.email == userEmail;
      
      // Only superadmins can write roles
      allow write: if isSignedIn() && isSuperAdmin();
    }

    // --- Gallery images ---
    match /gallery/{id} {
      allow read: if true;
      allow create, update, delete: if isSignedIn();
    }

    // --- Program Galleries (per program) ---
    match /programGalleries/{programId} {
      allow read: if true;
      allow create, update, delete: if isSignedIn() && isAdmin();
    }

    // --- Testimonial Galleries ---
    match /testimonialGalleries/{galleryId} {
      allow read: if true;
      allow create, update, delete: if isSignedIn() && isAdmin();
    }

    // --- Tour Requests (ECSE Classroom Tours) ---
    match /tourRequests/{id} {
      allow create: if request.resource.data.keys().hasAll([
        'parentName', 'childName', 'childDOB', 'childAge', 'lea', 'phone', 
        'address', 'email', 'needsTransportation', 'attendingPreschool', 
        'primaryLanguage', 'services', 'preferredTime', 'submittedAt', 'status'
      ]) && 
      request.resource.data.parentName is string &&
      request.resource.data.childName is string &&
      request.resource.data.childDOB is string &&
      request.resource.data.childAge is number &&
      request.resource.data.lea is string &&
      request.resource.data.phone is string &&
      request.resource.data.address is string &&
      request.resource.data.email is string &&
      request.resource.data.needsTransportation is bool &&
      request.resource.data.attendingPreschool is bool &&
      request.resource.data.primaryLanguage is string &&
      request.resource.data.services is list &&
      request.resource.data.preferredTime is string &&
      request.resource.data.submittedAt is timestamp &&
      request.resource.data.status is string;
      
      allow read, update, delete: if isSignedIn() && isAdmin();
    }

    // --- Testimonials ---
    match /testimonials/{id} {
      allow read: if true;
      allow create, update, delete: if isSignedIn();
    }

    // --- Job Openings ---
    match /jobs/{id} {
      allow read: if true;
      allow create, update, delete: if isSignedIn() && isAdmin();
    }

    // --- Job Applications ---
    match /job_applications/{id} {
      allow read: if isSignedIn() && isAdmin();
      allow create: if true;
      allow update, delete: if isSignedIn() && isAdmin();
    }

    // --- Blog Posts ---
    match /blog/{id} {
      allow read: if true;
      allow create, update, delete: if isSignedIn() && isAdmin();
    }

    // --- Applications (Job Applications) ---
    match /applications/{id} {
      allow read: if isSignedIn() && isAdmin();
      allow create: if true;
      allow update, delete: if isSignedIn() && isAdmin();
    }

    // --- Roles collection ---
    match /roles/{email} {
      allow read, create, update, delete: if isSignedIn();
    }

    // Deny everything else
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
